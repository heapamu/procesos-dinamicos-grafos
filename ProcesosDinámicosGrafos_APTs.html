# Calculo su matriz de adyacencia mediante la funciÃ³n definida en la PrÃ¡ctica del Tema 1 de TeorÃ­a de Grafos
def matriz_adyacencia(grafo):
    '''
    Crea la matriz de adyacencia a partir del grafo de entrada, ya sea dirigido o no dirigido.
    
    ParÃ¡metros
    ----------
    grafo : diccionario
        Diccionario que contiene el grafo de entrada.
        
    Returns
    -------
    MG : array
        Matriz de adyacencia del grafo de entrada.
    '''
    
    # Inicializamos una matriz nxn de ceros (siendo n el nÃºmero de nodos del grafo):
    MG = np.zeros((len(grafo), len(grafo)), dtype=int)
    
    for n,k in enumerate(grafo):
        for i in grafo[k]:
            MG[n, i-1] = 1
    
    return MG

Vamos a implementar ahora el sistema de ecuaciones diferenciales que implementa el modelo que nos da el estado de infecciÃ³n de cada uno de los nodos del grafo en cada instante de tiempo:

# Implementamos la funciÃ³n que devuelve dI/dt, siendo I el vector de funciones [I1, I2, ..., In] que 
# estamos buscando:
def modelo_infeccion(I, t, beta, gamma, phi):
    # Expresamos las ecuaciones del modelo
    dIdt = []
    for i in range(len(MG)):
        dIdt.append(beta * MG[i,:] @ I * (phi - I[i]) - gamma * I[i])
    
    return dIdt
